    SYM( at: ?int, txt: ?str) :SYM
    Summarize a stream of non-numerics.

    local SYM = obj("SYM", function(i,at,txt)
  i.at, i.txt, i.n, i.kept =  at or 0, txt or "", 0, {} end)

    Add n count to i.kept[n] .

    function SYM.add(i,x,n)
  if x ~= "?" then 
    i.n = i.n+1
    i.kept[x] = (n or 1) + (i.kept[x] or 0) end end

   return ((i.kept[x] or 0)+the.m*prior) / (i.n+the.m) end

    merge( i: SYM, j: SYM):SYM
    combine two symbols

    function SYM.merge(i,j,     k)
  k = i:clone()
  for _,kept in pairs{i.kept, j.kept} do
    for x,n in pairs(kept) do k:add(x,n) end end
  return k end

    div( i: SYM):tab
    Return diversity of a column (its tendency not to be a its central tendency).

    function SYM.div(i,p)
  local ent, log = 0, function(x) return math.log(x,2) end
  for x,n in pairs(i.kept) do if n > 0 then ent=ent - n/i.n*log(n/i.n) end end
  return ent end
